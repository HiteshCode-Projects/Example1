             let notes = []  //Array - To Store Data of Notes

             function addNote(){
                //Logic - Get Data from user(Input) and Add in my Ul Tag(List)

               let input = document.getElementById("noteInput");
               let message = document.getElementById("message")

               let value = input.value;

               if(value === ""){
                message.innerHTML = "Please write something";
                return;
               }

               //ArrayName.push(Value Which I am Adding)
               notes.push(value)
               console.log(notes)

               input.value = "";
               message.innerHTML = ""   


             }




             //Theme Change Task
             function toggleMode(){

             let body = document.getElementById("body")   //Body tag Selected

                 body.classList.remove("dark")

             }else{
                 body.classList.add("dark")
             }
